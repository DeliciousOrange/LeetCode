public class Solution {
    public int RemoveDuplicates(int[] nums) {
            int i = nums.Length > 0 ? 1 : 0;
            for (int j = 1; j < nums.Length; j++) {
                if (nums[j] != nums[i]) {
                    nums[i++] = nums[j];
                }
            }
            return i;   
        }
}
